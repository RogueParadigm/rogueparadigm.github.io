Multiplayer Guide

Designing multiplayer-safe GenAI systems with ElysGenAI.

Core Principle‚Äã

üéØ Audio processing is CLIENT-SIDE ONLY

Never replicate raw audio over the network. Only replicate text results and metadata.

Architecture Rules‚Äã

‚úÖ What Runs on Client‚Äã

// CLIENT ONLY
- Audio capture (microphone)
- Speech-to-text processing
- LLM generation (for player assistance)
- Voice activity detection
- Audio buffering

‚úÖ What Runs on Server‚Äã

// SERVER
- Text processing (from STT results)
- Intent parsing
- Command validation
- Game state changes
- NPC dialogue generation (LLM)

‚ùå Never Replicate‚Äã

// DO NOT REPLICATE
- Raw audio data (TArray<float>)
- Audio buffers
- Model inference results (internal)
- Backend instances

‚úÖ Always Replicate‚Äã

// DO REPLICATE
- Transcribed text (FString)
- Parsed intents
- Command confirmations
- NPC responses

Network Modes‚Äã

Check network mode before processing audio:

void UERP_STTComponent::BeginPlay()
{
    Super::BeginPlay();
    
    // Only process audio on owning client
    if (GetNetMode() == NM_DedicatedServer)
    {
        UE_LOG(LogElysSTT, Warning, 
            TEXT("STT Component on dedicated server - audio capture disabled"));
        return;
    }
    
    if (!GetOwner()->HasAuthority() || GetOwner()->IsLocallyControlled())
    {
        // Safe to capture audio
        StartListening();
    }
}

Voice Command Pattern‚Äã

Client-Side Processing‚Äã

// PlayerController.h
UCLASS()
class AMyPlayerController : public APlayerController
{
    GENERATED_BODY()
    
    UPROPERTY(VisibleAnywhere)
    UERP_STTComponent* STTComponent;
    
    UFUNCTION(Server, Reliable, WithValidation)
    void ServerExecuteVoiceCommand(const FString& Command, float Confidence);
    
    UFUNCTION()
    void OnTranscriptionReceived(const FElysSTTResult& Result);
};

// PlayerController.cpp
void AMyPlayerController::BeginPlay()
{
    Super::BeginPlay();
    
    // Only on owning client
    if (IsLocalController())
    {
        STTComponent->OnTranscriptionComplete.AddDynamic(
            this, &AMyPlayerController::OnTranscriptionReceived);
        STTComponent->StartListening();
    }
}

void AMyPlayerController::OnTranscriptionReceived(const FElysSTTResult& Result)
{
    // Client-side: transcribe speech
    if (Result.Confidence > 0.7f)
    {
        // Send text to server
        ServerExecuteVoiceCommand(Result.TranscribedText, Result.Confidence);
    }
}

// Server RPC
void AMyPlayerController::ServerExecuteVoiceCommand_Implementation(
    const FString& Command, float Confidence)
{
    // Server-side: validate and execute
    if (Command.Contains(TEXT("attack")))
    {
        GetPawn()->PerformAttack();
    }
    else if (Command.Contains(TEXT("reload")))
    {
        GetPawn()->ReloadWeapon();
    }
}

bool AMyPlayerController::ServerExecuteVoiceCommand_Validation(
    const FString& Command, float Confidence)
{
    // Prevent abuse
    return Confidence > 0.5f && Command.Len() < 256;
}

NPC Dialogue Pattern‚Äã

Server-Side LLM‚Äã

NPCs should generate dialogue on the server to ensure consistency:

// NPC Character.h
UCLASS()
class ANPCCharacter : public ACharacter
{
    GENERATED_BODY()
    
    UPROPERTY(VisibleAnywhere)
    UERP_LLMComponent* LLMComponent;
    
    UPROPERTY(ReplicatedUsing=OnRep_CurrentDialogue)
    FString CurrentDialogue;
    
    UFUNCTION()
    void OnRep_CurrentDialogue();
    
    UFUNCTION(NetMulticast, Reliable)
    void MulticastDisplayDialogue(const FString& Dialogue);
    
    void GenerateResponse(const FString& PlayerMessage);
};

// NPC Character.cpp
void ANPCCharacter::BeginPlay()
{
    Super::BeginPlay();
    
    // Only on server
    if (HasAuthority())
    {
        LLMComponent->OnGenerationComplete.AddDynamic(
            this, &ANPCCharacter::OnDialogueGenerated);
    }
}

void ANPCCharacter::GenerateResponse(const FString& PlayerMessage)
{
    if (!HasAuthority()) return;
    
    // Server generates response
    LLMComponent->SendMessage(PlayerMessage);
}

void ANPCCharacter::OnDialogueGenerated(const FElysLLMResult& Result)
{
    CurrentDialogue = Result.GeneratedText;
    
    // Replicate to all clients
    MulticastDisplayDialogue(CurrentDialogue);
}

void ANPCCharacter::MulticastDisplayDialogue_Implementation(const FString& Dialogue)
{
    // All clients display dialogue UI
    ShowDialogueBubble(Dialogue);
}

Player Asks NPC (Full Flow)‚Äã

// PlayerController sends voice command to server
void AMyPlayerController::OnTranscriptionReceived(const FElysSTTResult& Result)
{
    // Client: "Where is the blacksmith?"
    ServerAskNPC(TargetedNPC, Result.TranscribedText);
}

// Server receives text, tells NPC to respond
void AMyPlayerController::ServerAskNPC_Implementation(ANPCCharacter* NPC, const FString& Question)
{
    if (NPC)
    {
        NPC->GenerateResponse(Question);
    }
}

// NPC generates response on server
void ANPCCharacter::GenerateResponse(const FString& Question)
{
    // Server-side LLM generation
    LLMComponent->SendMessage(Question);
}

// NPC replicates response to all clients
void ANPCCharacter::OnDialogueGenerated(const FElysLLMResult& Result)
{
    MulticastDisplayDialogue(Result.GeneratedText);
}

Voice Chat Pattern‚Äã

Transcription + Audio Streaming‚Äã

Hybrid approach: stream audio for voice chat, but also transcribe for moderation:

// VoiceChatComponent.h
UCLASS()
class UVoiceChatComponentERP : public UActorComponent, public IERP_AudioConsumer
{
    GENERATED_BODY()
    
public:
    // IERP_AudioConsumer
    virtual void OnAudioDataReceived_Implementation(const FElysAudioBuffer& Buffer) override;
    
    UPROPERTY(VisibleAnywhere)
    UERP_STTComponent* STTComponent;
    
    UFUNCTION(Server, Unreliable, WithValidation)
    void ServerStreamAudio(const TArray<uint8>& CompressedAudio);
    
    UFUNCTION(Server, Reliable, WithValidation)
    void ServerReportTranscription(const FString& Text, float Confidence);
};

// VoiceChatComponent.cpp
void UVoiceChatComponentERP::OnAudioDataReceived_Implementation(const FElysAudioBuffer& Buffer)
{
    // 1. Stream compressed audio to server (Opus encoding)
    TArray<uint8> CompressedAudio = CompressAudioOpus(Buffer.AudioData);
    ServerStreamAudio(CompressedAudio);
    
    // 2. Also transcribe for moderation (no audio sent)
    STTComponent->ProcessAudioBuffer(Buffer);
}

void UVoiceChatComponentERP::OnTranscriptionComplete(const FElysSTTResult& Result)
{
    // Send transcription to server for content moderation
    ServerReportTranscription(Result.TranscribedText, Result.Confidence);
}

// Server checks for toxic content
void UVoiceChatComponentERP::ServerReportTranscription_Implementation(
    const FString& Text, float Confidence)
{
    if (ContainsToxicContent(Text))
    {
        MutePlayer(GetOwner());
        NotifyModerators(Text);
    }
}

Bandwidth Considerations‚Äã

Audio Replication (DON'T DO THIS)‚Äã

// ‚ùå NEVER DO THIS - Wastes bandwidth
UPROPERTY(Replicated)
TArray<float> AudioBuffer; // 16000 samples/sec * 4 bytes = 64KB/sec

// ‚ùå BAD - Replicating raw audio
UFUNCTION(Server, Unreliable)
void ServerSendAudio(const TArray<float>& Audio);

Text Replication (DO THIS)‚Äã

// ‚úÖ GOOD - Efficient text replication
UPROPERTY(Replicated)
FString TranscribedText; // ~50 bytes average

// ‚úÖ GOOD - Minimal bandwidth
UFUNCTION(Server, Reliable)
void ServerSendCommand(const FString& Command);

Example: Multiplayer Voice Commands‚Äã

Setup‚Äã

// GameMode.h
UCLASS()
class AMyGameMode : public AGameModeBase
{
    UFUNCTION()
    void HandlePlayerVoiceCommand(APlayerController* Player, const FString& Command);
};

// PlayerController.h
UCLASS()
class AMyPlayerController : public APlayerController
{
    UPROPERTY(VisibleAnywhere)
    UERP_STTComponent* STTComponent;
    
    UFUNCTION(Server, Reliable, WithValidation)
    void ServerVoiceCommand(const FString& Command, float Confidence);
};

Implementation‚Äã

void AMyPlayerController::BeginPlay()
{
    Super::BeginPlay();
    
    if (IsLocalController())
    {
        STTComponent->OnTranscriptionComplete.AddDynamic(
            this, &AMyPlayerController::OnVoiceInput);
        STTComponent->StartListening();
    }
}

void AMyPlayerController::OnVoiceInput(const FElysSTTResult& Result)
{
    // Client processes speech
    if (Result.Confidence > 0.7f)
    {
        // Send text command to server
        ServerVoiceCommand(Result.TranscribedText, Result.Confidence);
        
        // Local feedback
        ClientMessage(FString::Printf(TEXT("You said: %s"), 
            *Result.TranscribedText));
    }
}

void AMyPlayerController::ServerVoiceCommand_Implementation(
    const FString& Command, float Confidence)
{
    // Server validates and executes
    AMyGameMode* GM = Cast<AMyGameMode>(GetWorld()->GetAuthGameMode());
    if (GM)
    {
        GM->HandlePlayerVoiceCommand(this, Command);
    }
}

bool AMyPlayerController::ServerVoiceCommand_Validation(
    const FString& Command, float Confidence)
{
    // Anti-cheat checks
    return Command.Len() > 0 && 
           Command.Len() < 256 && 
           Confidence >= 0.0f && 
           Confidence <= 1.0f;
}

void AMyGameMode::HandlePlayerVoiceCommand(
    APlayerController* Player, const FString& Command)
{
    // Parse and execute command
    if (Command.Contains(TEXT("need ammo")))
    {
        SpawnAmmoNearPlayer(Player);
    }
    else if (Command.Contains(TEXT("regroup")))
    {
        BroadcastRegroupCommand(Player->GetTeamNum());
    }
}

Security Considerations‚Äã

Validate All Client Input‚Äã

bool AMyPlayerController::ServerVoiceCommand_Validation(
    const FString& Command, float Confidence)
{
    // Length check
    if (Command.Len() > 256) return false;
    
    // Confidence range check
    if (Confidence < 0.0f || Confidence > 1.0f) return false;
    
    // Rate limiting
    float TimeSinceLastCommand = GetWorld()->GetTimeSeconds() - LastCommandTime;
    if (TimeSinceLastCommand < MinCommandInterval) return false;
    
    // Command whitelist (optional)
    static const TArray<FString> AllowedCommands = {
        TEXT("attack"), TEXT("defend"), TEXT("regroup"), TEXT("need ammo")
    };
    
    bool bIsAllowed = false;
    for (const FString& Allowed : AllowedCommands)
    {
        if (Command.Contains(Allowed))
        {
            bIsAllowed = true;
            break;
        }
    }
    
    return bIsAllowed;
}

Prevent Replay Attacks‚Äã

// Add timestamp to commands
UFUNCTION(Server, Reliable, WithValidation)
void ServerVoiceCommand(const FString& Command, float Confidence, float Timestamp);

bool AMyPlayerController::ServerVoiceCommand_Validation(
    const FString& Command, float Confidence, float Timestamp)
{
    // Reject old commands (>1 second)
    float ServerTime = GetWorld()->GetTimeSeconds();
    if (FMath::Abs(ServerTime - Timestamp) > 1.0f) return false;
    
    return true;
}

Summary‚Äã

‚úÖ Best Practices‚Äã

  • Audio processing on client only
  • Replicate text results, not audio
  • Use Server RPCs for commands
  • Validate all client input
  • Rate limit command frequency
  • Check network mode before audio capture

‚ùå Anti-Patterns‚Äã

  • Replicating raw audio buffers
  • Processing audio on dedicated server
  • Trusting client-side validation
  • No rate limiting
  • Unvalidated text commands
