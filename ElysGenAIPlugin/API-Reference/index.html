UElysGenAISettings‚Äã

Project settings for ElysGenAI Framework.

Path: Project Settings ‚Üí Elys GenAI Framework

UCLASS(Config=Game, DefaultConfig)
class ELYSGENA ICORE_API UElysGenAISettings : public UDeveloperSettings
{
    // Audio Settings
    UPROPERTY(Config, EditAnywhere, Category="Audio")
    int32 DefaultSampleRate = 16000;
    
    UPROPERTY(Config, EditAnywhere, Category="Audio")
    int32 DefaultNumChannels = 1;
    
    UPROPERTY(Config, EditAnywhere, Category="Audio")
    int32 DefaultBitsPerSample = 16;
    
    // STT Settings
    UPROPERTY(Config, EditAnywhere, Category="STT")
    EElysSTTBackendType STTBackend = EElysSTTBackendType::Whisper;
    
    UPROPERTY(Config, EditAnywhere, Category="STT")
    FString STTModelPath;
    
    UPROPERTY(Config, EditAnywhere, Category="STT")
    int32 STTNumThreads = 4;
    
    // LLM Settings
    UPROPERTY(Config, EditAnywhere, Category="LLM")
    EElysLLMBackendType LLMBackend = EElysLLMBackendType::LlamaCpp;
    
    UPROPERTY(Config, EditAnywhere, Category="LLM")
    FString LLMModelPath;
    
    UPROPERTY(Config, EditAnywhere, Category="LLM")
    int32 LLMContextLength = 4096;
};

class IERP_GenAIService
{
public:
    // Lifecycle
    virtual bool Initialize() = 0;

    virtual bool IsReady() const = 0;
    
    // State
    virtual EElysGenAIServiceState GetProcessingState() const = 0;
    
    // Error handling
    virtual EElysGenAIErrorCode GetLastError() const = 0;
    virtual FString GetLastErrorMessage() const = 0;
    virtual void ClearError() = 0;
};

FElysAudioFormat‚Äã

Audio format specification.

USTRUCT(BlueprintType)
struct FElysAudioFormat
{
    GENERATED_BODY()
    
    UPROPERTY(BlueprintReadWrite)
    int32 SampleRate = 16000;  // Hz
    
    UPROPERTY(BlueprintReadWrite)
    int32 NumChannels = 1;  // 1=Mono, 2=Stereo
    
    UPROPERTY(BlueprintReadWrite)
    int32 BitsPerSample = 16;  // 8, 16, 24, 32
};

class ELYSGENAISTT_API UERP_STTComponent : public UActorComponent
{
    GENERATED_BODY()

    UFUNCTION(BlueprintCallable, Category="ElysGenAI|STT")

    UFUNCTION(BlueprintCallable, Category="ElysGenAI|STT")
    void SetMinConfidence(float MinConfidence);
    
    // Properties
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="ElysGenAI|STT")
    bool bAutoStartListening = false;
    
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="ElysGenAI|STT")

UCLASS(BlueprintType, ClassGroup=(ElysGenAI), meta=(BlueprintSpawnableComponent))
class ELYSGENAILLM_API UERP_LLMComponent : public UActorComponent
{
    GENERATED_BODY()
    
public:
    // Generation
    UFUNCTION(BlueprintCallable, Category="ElysGenAI|LLM")
    void SendMessage(const FString& Message);
    
    UFUNCTION(BlueprintCallable, Category="ElysGenAI|LLM")
    void SendMessageWithRole(const FString& Message, const FString& Role);
    
    // Configuration
    UFUNCTION(BlueprintCallable, Category="ElysGenAI|LLM")
    void SetSystemPrompt(const FString& Prompt);
    
    UFUNCTION(BlueprintCallable, Category="ElysGenAI|LLM")
    void SetTemperature(float Temperature);
    
    UFUNCTION(BlueprintCallable, Category="ElysGenAI|LLM")
    void SetMaxTokens(int32 MaxTokens);
    
    // History management
    UFUNCTION(BlueprintCallable, Category="ElysGenAI|LLM")
    void ClearHistory();
    
    UFUNCTION(BlueprintPure, Category="ElysGenAI|LLM")
    TArray<FElysLLMMessage> GetHistory() const;
    
    // Properties
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="ElysGenAI|LLM")
    FString SystemPrompt = TEXT("You are a helpful assistant.");
    
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="ElysGenAI|LLM")
    int32 MaxTokens = 256;
    
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="ElysGenAI|LLM")
    float Temperature = 0.7f;
    
    UPROPERTY(EditAnywhere, BlueprintReadWrite, Category="ElysGenAI|LLM")
    int32 MaxHistoryMessages = 20;
    
    // Events
    UPROPERTY(BlueprintAssignable, Category="ElysGenAI|LLM")
    FElysLLMResultDelegate OnGenerationComplete;
    
    UPROPERTY(BlueprintAssignable, Category="ElysGenAI|LLM")
    FElysLLMTokenDelegate OnTokenGenerated;
};
