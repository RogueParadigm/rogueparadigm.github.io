Advanced Customization

Extending ElysGenAI with custom backends, contexts, and consumers.

Custom STT Backend‚Äã

Implement IERP_STTBackend to support alternative STT engines:

// MyCustomSTTBackend.h
#pragma once

#include "CoreMinimal.h"
#include "IERP_STTBackend.h"
#include "MyCustomSTTBackend.generated.h"

UCLASS()
class UMyCustomSTTBackendERP : public UObject, public IERP_STTBackend
{
    GENERATED_BODY()
    
public:
    // IERP_GenAIService interface
    virtual bool Initialize() override;
    virtual void Shutdown() override;
    virtual bool IsReady() const override;
    virtual EElysGenAIServiceState GetProcessingState() const override;
    virtual EElysGenAIErrorCode GetLastError() const override;
    virtual FString GetLastErrorMessage() const override;
    virtual void ClearError() override;
    
    // IERP_STTBackend interface
    virtual bool LoadModel(const FString& ModelPath) override;
    virtual void UnloadModel() override;
    virtual bool IsModelLoaded() const override;
    virtual void SetLanguage(const FString& LanguageCode) override;
    virtual void SetSampleRate(int32 SampleRate) override;
    virtual FElysSTTResult ProcessAudio(const TArray<float>& AudioData) override;
    virtual void ProcessAudioAsync(const TArray<float>& AudioData, 
                                    const FElysSTTResultDelegate& Callback) override;
    
private:
    bool bIsInitialized = false;
    bool bModelLoaded = false;
    FString CurrentLanguage;
    int32 CurrentSampleRate;
    
    // Your custom STT engine instance
    void* CustomSTTEngine;
};

// MyCustomSTTBackend.cpp
bool UMyCustomSTTBackendERP::Initialize()
{
    // Initialize your STT engine
    CustomSTTEngine = CreateYourSTTEngine();
    bIsInitialized = (CustomSTTEngine != nullptr);
    return bIsInitialized;
}

bool UMyCustomSTTBackendERP::LoadModel(const FString& ModelPath)
{
    if (!bIsInitialized) return false;
    
    // Load your model
    bModelLoaded = YourSTTEngine_LoadModel(CustomSTTEngine, ModelPath);
    return bModelLoaded;
}

FElysSTTResult UMyCustomSTTBackendERP::ProcessAudio(const TArray<float>& AudioData)
{
    FElysSTTResult Result;
    
    if (!IsReady())
    {
        Result.TranscribedText = TEXT("");
        Result.Confidence = 0.0f;
        return Result;
    }
    
    // Call your STT engine
    FString Transcription = YourSTTEngine_Transcribe(CustomSTTEngine, AudioData);
    float Confidence = YourSTTEngine_GetConfidence(CustomSTTEngine);
    
    Result.TranscribedText = Transcription;
    Result.Confidence = Confidence;
    Result.Language = CurrentLanguage;
    Result.bIsFinal = true;
    
    return Result;
}

Register Custom Backend‚Äã

// In your subsystem
TScriptInterface<IERP_STTBackend> UERP_STTSubsystem::CreateBackend()
{
    // Create custom backend instead of Whisper
    UMyCustomSTTBackendERP* CustomBackend = NewObject<UMyCustomSTTBackendERP>(this);
    
    TScriptInterface<IERP_STTBackend> BackendInterface;
    BackendInterface.SetObject(CustomBackend);
    BackendInterface.SetInterface(Cast<IERP_STTBackend>(CustomBackend));
    
    return BackendInterface;
}

Custom LLM Backend‚Äã

Implement IERP_LLMBackend for alternative language models:

// MyCustomLLMBackend.h
UCLASS()
class UMyCustomLLMBackendERP : public UObject, public IERP_LLMBackend
{
    GENERATED_BODY()
    
public:
    // IERP_GenAIService interface
    virtual bool Initialize() override;
    virtual void Shutdown() override;
    virtual bool IsReady() const override;
    // ... other service methods
    
    // IERP_LLMBackend interface
    virtual bool LoadModel(const FString& ModelPath, const FElysLLMConfig& Config) override;
    virtual void UnloadModel() override;
    virtual bool IsModelLoaded() const override;
    virtual FElysLLMResult Generate(const FString& Prompt, const FElysLLMConfig& Config) override;
    virtual void GenerateAsync(const FString& Prompt, 
                               const FElysLLMConfig& Config,
                               const FElysLLMResultDelegate& Callback) override;
    virtual void GenerateStreaming(const FString& Prompt,
                                   const FElysLLMConfig& Config,
                                   const FElysLLMTokenDelegate& TokenCallback,
                                   const FElysLLMResultDelegate& CompleteCallback) override;
    
private:
    void* CustomLLMEngine;
    bool bIsInitialized = false;
    bool bModelLoaded = false;
};

// MyCustomLLMBackend.cpp
FElysLLMResult UMyCustomLLMBackendERP::Generate(
    const FString& Prompt, const FElysLLMConfig& Config)
{
    FElysLLMResult Result;
    
    if (!IsReady())
    {
        Result.GeneratedText = TEXT("");
        Result.TokenCount = 0;
        Result.FinishReason = EElysLLMFinishReason::Error;
        return Result;
    }
    
    // Generate with your LLM
    FString Generated = YourLLM_Generate(CustomLLMEngine, Prompt, Config);
    int32 Tokens = YourLLM_GetTokenCount(CustomLLMEngine);
    
    Result.GeneratedText = Generated;
    Result.TokenCount = Tokens;
    Result.FinishReason = EElysLLMFinishReason::Completed;
    
    return Result;
}

Custom Audio Consumer‚Äã

Create custom audio consumers for specialized processing:

// AudioRecorderComponent.h
UCLASS()
class UAudioRecorderComponent : public UActorComponent, 
                                  public IERP_AudioConsumer
{
    GENERATED_BODY()
    
public:
    // IERP_AudioConsumer interface
    virtual void OnAudioDataReceived_Implementation(
        const FElysAudioBuffer& Buffer) override;
    
    virtual FString GetConsumerName_Implementation() const override
    {
        return TEXT("AudioRecorder");
    }
    
    UFUNCTION(BlueprintCallable, Category="Audio")
    void StartRecording();
    
    UFUNCTION(BlueprintCallable, Category="Audio")
    void StopRecording();
    
    UFUNCTION(BlueprintCallable, Category="Audio")
    bool SaveToFile(const FString& Filename);
    
private:
    bool bIsRecording = false;
    TArray<float> RecordedAudio;
    FElysAudioFormat RecordingFormat;
};

// AudioRecorderComponent.cpp
void UAudioRecorderComponent::BeginPlay()
{
    Super::BeginPlay();
    
    // Register with audio subsystem
    if (UERP_AudioCaptureSubsystem* AudioSubsystem = 
        GetGameInstance()->GetSubsystem<UERP_AudioCaptureSubsystem>())
    {
        AudioSubsystem->RegisterConsumer(this);
    }
}

void UAudioRecorderComponent::OnAudioDataReceived_Implementation(
    const FElysAudioBuffer& Buffer)
{
    if (bIsRecording)
    {
        RecordedAudio.Append(Buffer.AudioData);
        RecordingFormat = Buffer.Format;
    }
}

void UAudioRecorderComponent::StartRecording()
{
    RecordedAudio.Empty();
    bIsRecording = true;
}

void UAudioRecorderComponent::StopRecording()
{
    bIsRecording = false;
}

bool UAudioRecorderComponent::SaveToFile(const FString& Filename)
{
    // Convert float32 to int16
    TArray<int16> PCMData;
    for (float Sample : RecordedAudio)
    {
        PCMData.Add(static_cast<int16>(Sample * 32767.0f));
    }
    
    // Write WAV file
    return WriteWAVFile(Filename, PCMData, RecordingFormat);
}

using UnrealBuildTool;

public class ElysGenAI_MyModule : ModuleRules
{
    public ElysGenAI_MyModule(ReadOnlyTargetRules Target) : base(Target)
    {
        PCHUsage = ModuleRules.PCHUsageMode.UseExplicitOrSharedPCHs;
        
        PublicDependencyModuleNames.AddRange(new string[]

            "Core",
            "CoreUObject",
            "Engine",

}

Wwise Integration‚Äã

UCLASS()
class UElysGenAI_WwiseComponent : public UActorComponent, 
                                   public IERP_AudioConsumer
{
    GENERATED_BODY()
    
public:
    virtual void OnAudioDataReceived_Implementation(
        const FElysAudioBuffer& Buffer) override
    {
        // Send audio to Wwise for processing
        PostAudioToWwise(Buffer.AudioData);
        
        // Apply Wwise effects
        TArray<float> ProcessedAudio = GetWwiseOutput();
        
        // Forward to other consumers
        BroadcastProcessedAudio(ProcessedAudio);
    }
};

Metasounds Integration‚Äã

// Create Metasound source from STT events
UCLASS()
class UElysGenAI_MetasoundComponent : public UActorComponent
{
    GENERATED_BODY()
    
    UPROPERTY()
    UMetaSoundSource* STTTriggeredSound;
    
    UFUNCTION()
    void OnTranscription(const FElysSTTResult& Result)
    {
        // Trigger Metasound on transcription
        if (STTTriggeredSound)
        {
            UAudioComponent* AudioComp = 
                UGameplayStatics::SpawnSound2D(this, STTTriggeredSound);
            
            // Set parameters based on confidence
            AudioComp->SetFloatParameter(
                TEXT("Confidence"), Result.Confidence);
        }
    }
};

Unit Tests‚Äã

// MyCustomBackendTest.cpp
#include "Misc/AutomationTest.h"
#include "MyCustomSTTBackendERP.h"

IMPLEMENT_SIMPLE_AUTOMATION_TEST(
    FMyCustomSTTBackendTest,
    "ElysGenAI.STT.CustomBackend",
    EAutomationTestFlags::ApplicationContextMask | 
    EAutomationTestFlags::ProductFilter)

bool FMyCustomSTTBackendTest::RunTest(const FString& Parameters)
{
    // Create backend
    UMyCustomSTTBackendERP* Backend = NewObject<UMyCustomSTTBackendERP>();
    
    // Test initialization
    TestTrue(TEXT("Backend initializes"), Backend->Initialize());
    TestTrue(TEXT("Backend is ready"), Backend->IsReady());
    
    // Test model loading
    FString ModelPath = TEXT("Path/To/Test/Model.bin");
    TestTrue(TEXT("Model loads"), Backend->LoadModel(ModelPath));
    TestTrue(TEXT("Model is loaded"), Backend->IsModelLoaded());
    
    // Test audio processing
    TArray<float> TestAudio;
    TestAudio.SetNum(16000);  // 1 second at 16kHz
    
    FElysSTTResult Result = Backend->ProcessAudio(TestAudio);
    TestNotNull(TEXT("Result is valid"), Result.TranscribedText);
    
    // Cleanup
    Backend->Shutdown();
    
    return true;
}
